from bs4 import BeautifulSoup
import re
from pdfminer.pdfinterp import PDFResourceManager, PDFPageInterpreter
from pdfminer.converter import TextConverter
from pdfminer.layout import LAParams
from pdfminer.pdfpage import PDFPage

from io import StringIO

import html2text
# write pdf to text
def read_pdf(pdf_file):
    """
    Reads a pdf file and returns its text

    :param pdf_file: path to pdf file
    :return: text in pdf file

    """
    resource_mgr = PDFResourceManager()
    retstr = StringIO()
    codec = 'utf-8'
    laparams = LAParams()
    device = TextConverter(resource_mgr, retstr, laparams=laparams)
    interpreter = PDFPageInterpreter(resource_mgr, device)
    fp = open(pdf_file, 'rb')
    maxpages = 0
    caching = True
    pagenos = set()
    
    for page in PDFPage.get_pages(fp, pagenos, 
                                  maxpages=maxpages, 
                                  caching=caching, 
                                  check_extractable=True):
        interpreter.process_page(page)

    result_str = retstr.getvalue()
    
    fp.close()
    device.close()
    retstr.close()
    
    return result_str

def string_to_txt(string, txt_file):

    with open(txt_file, 'w') as f:
        f.write(string)


def pdf_to_txt(pdf_file, txt_file):

    string = read_pdf(pdf_file)
    string_to_txt(string, txt_file)

def get_reference_text(ref_html):
        """Retreives text from html judgement

        Args:
            ref_html (str): judgement in html format

        Returns:
            str: judgement in text format
        """
        #vineet-irfan
        soup = BeautifulSoup(ref_html, features="html.parser")
        meta_data = ""
        # write directly to summary and remove (extract()) these divs from consideration
        for div_class in ['docsource_main', 'doc_title', 'doc_citations', 'doc_author', 'doc_bench']:
            find_div_class = soup.find('div', {'class': div_class})
            if find_div_class:
                meta_data += div_class + ": " + find_div_class.extract().get_text() + "\n"
            # TODO: write directy to summary
        meta_data += "\n" + ('*' * 50) + "\n\n"

        # similarly for the <pre> tag
        pre = soup.find('pre')
        if pre:
            pass
            # print(pre.extract().get_text())

        # remove all hidden text spans
        for hidden_text_span in soup.find_all('span', {'class': 'hidden_text'}):
            hidden_text_span.extract()

        reference_text_paras = []
        for para in soup.find_all('p'):
            para_text = para.get_text()
            # replace multiple whitespace characters (including \n) with a single space
            para_text = re.sub(r'\s+', ' ', para_text)
            para_text = para_text.strip()
            # remove para numbers like '12.'
            para_text = para_text.lstrip('0123456789.')
            reference_text_paras.append(para_text.strip())
        reference_text_paras

        return ' '.join(reference_text_paras), meta_data
